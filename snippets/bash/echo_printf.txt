#!/bin/bash

# Basic echo
echo "Simple text"
echo -n "No newline"
echo " - continued"

# Echo with escape sequences
echo -e "Line 1\nLine 2\tTabbed"
echo -e "\033[31mRed text\033[0m"  # Colors

# Printf formatting
printf "Name: %-10s Age: %3d\n" "Alice" 25
printf "Price: $%.2f\n" 19.99
printf "Hex: %x, Octal: %o\n" 255 255

# Printf with variables
name="Bob"
score=87.5
printf "Student: %s scored %.1f%%\n" "$name" "$score"
