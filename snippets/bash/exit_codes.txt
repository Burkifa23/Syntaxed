#!/bin/bash

# Function that may fail
check_file() {
    if [[ -f "$1" ]]; then
        echo "File exists: $1"
        return 0  # Success
    else
        echo "File not found: $1" >&2
        return 1  # Failure
    fi
}

# Check exit codes
check_file "/etc/passwd"
if [[ $? -eq 0 ]]; then
    echo "Previous command succeeded"
fi
