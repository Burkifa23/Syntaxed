#!/bin/bash

# Basic for loop
for i in 1 2 3 4 5; do
    echo "Number: $i"
done

# For loop with range
for i in {1..10}; do
    echo "Count: $i"
done

# For loop with step
for i in {0..20..2}; do
    echo "Even: $i"
done

# C-style for loop
for ((i=0; i<10; i++)); do
    echo "Index: $i"
done

# For loop with array
fruits=("apple" "banana" "cherry")
for fruit in "${fruits[@]}"; do
    echo "Fruit: $fruit"
done

# For loop with files
for file in *.txt; do
    if [[ -f $file ]]; then
        echo "Processing: $file"
        wc -l "$file"
    fi
done

# For loop with command output
for user in $(cut -d: -f1 /etc/passwd | head -5); do
    echo "User: $user"
done

# Nested for loops
for i in {1..3}; do
    for j in {a..c}; do
        echo "$i$j"
    done
done
