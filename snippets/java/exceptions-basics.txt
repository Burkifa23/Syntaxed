import java.io.File;
import java.io.FileReader;
import java.io.IOException;
import java.util.concurrent.TimeUnit;

public class Main {
    public static void main(String[] args) {
        System.out.println("--- 1. Unchecked Exception Demo (RuntimeException) ---");
        try {
            int result = calculateDivision(10, 0);
            System.out.println("Result: " + result);
        } catch (ArithmeticException e) {
            System.out.println("Error: Cannot divide by zero. Caught Unchecked Exception.");
            System.out.println("Message: " + e.getMessage());
        }

        System.out.println("\n--- 2. Checked Exception Demo (IOException) ---");
        FileReader reader = null;
        try {           
            reader = new FileReader(new File("non_existent_file.txt"));
            reader.read();
        } catch (IOException e) {
            System.out.println("Error: File operation failed. Caught Checked Exception.");
            System.out.println("Message: " + e.getMessage());
        } finally {
            System.out.println("Finally block executed: Resource cleanup attempt.");
            if (reader != null) {
                try {
                    reader.close();
                } catch (IOException e) {
                    
                }
            }
        }        
        System.out.println("\nProgram continues execution normally.");
    }

    public static int calculateDivision(int a, int b) {
        return a / b;
    }
}
