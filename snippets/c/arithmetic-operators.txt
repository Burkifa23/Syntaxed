#include <stdio.h>

int main() {
    int a = 15, b = 4, c = 7;
    int result;
    
    printf("Variables: a = %d, b = %d, c = %d\n\n", a, b, c);
    
    printf("a + b = %d\n", a + b);
    printf("a - b = %d\n", a - b);
    printf("a * b = %d\n", a * b);
    printf("a / b = %d\n", a / b);
    printf("a %% b = %d\n", a % b)
    
    // Float division
    printf("a / (float)b = %.2f\n", a / (float)b);
    
    return 0;
}
