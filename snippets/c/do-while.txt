#include <stdio.h>

int main() {

    int choice;

    printf("Simple menu system:\n");
    do {
        printf("\nMenu Options:\n");
        printf("1. Say Hello\n");
        printf("2. Calculate Square\n");
        printf("3. Exit\n");
        printf("Choose option: ");
        scanf("%d", &choice);
        
        switch (choice) {
            case 1:
                printf("Hello, World!\n");
                break;
            case 2:
                printf("Enter a number: ");
                scanf("%d", &n);
                printf("Square of %d is %d\n", n, n * n);
                break;
            case 3:
                printf("Goodbye!\n");
                break;
            default:
                printf("Invalid option!\n");
        }
    } while (choice != 3);

    return 0;

}
