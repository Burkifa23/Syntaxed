#include <stdio.h>
#include <string.h>

void reverse_string(char str[]);

int main(){

    // Reverse string
    char reverse_me[] = "ABCDEF";
    printf("Before reverse: %s\n", reverse_me);
    reverse_string(reverse_me);
    printf("After reverse: %s\n", reverse_me);
    return 0;

}

void reverse_string(char str[]) {
    int length = strlen(str);
    for (int i = 0; i < length / 2; i++) {
        char temp = str[i];
        str[i] = str[length - 1 - i];
        str[length - 1 - i] = temp;
    }
}
