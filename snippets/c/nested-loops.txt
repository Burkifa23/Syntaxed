#include <stdio.h>

int main() {
    int i,j;
    
    // Multiplication table
    printf("5x5 Multiplication Table:\n");
    for (i = 1; i <= 5; i++) {
        for (j = 1; j <= 5; j++) {
            printf("%3d", i * j);
        }
        printf("\n");
    }
    printf("\n");
    
    // Pattern printing
    printf("Star pattern:\n");
    for (i = 1; i <= 5; i++) {
        for (j = 1; j <= i; j++) {
            printf("* ");
        }
        printf("\n");
    }
    printf("\n");
    
    // Number pattern
    printf("Number pyramid:\n");
    for (i = 1; i <= 4; i++) {
        // Print spaces
        for (j = 1; j <= 4 - i; j++) {
            printf(" ");
        }
        // Print numbers
        for (j = 1; j <= i; j++) {
            printf("%d ", j);
        }
        printf("\n");
    }

    return 0;

}
