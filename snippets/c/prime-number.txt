#include <stdio.h>

int main() {
    int n, i;

    // Check if number is prime
    printf("Enter a number to check if prime: ");
    scanf("%d", &n);
    // Assume prime
    int is_prime = 1;
    
    if (n <= 1) {
        is_prime = 0;
    } else {
        for (i = 2; i * i <= n; i++) {
            if (n % i == 0) {
                is_prime = 0;
                break;
            }
        }
    }
    
    if (is_prime) {
        printf("%d is a prime number\n", n);
    } else {
        printf("%d is not a prime number\n", n);
    }
    
    return 0;
}
