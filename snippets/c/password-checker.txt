#include <stdio.h>
#include <string.h>

int main() {
    // Simple password checker
    char password[20];
    printf("Enter a password: ");
    scanf("%19s", password);
    
    int length = strlen(password);
    int has_digit = 0, has_upper = 0;
    
    for (int i = 0; i < length; i++) {
        if (password[i] >= '0' && password[i] <= '9') {
            has_digit = 1;
        }
        if (password[i] >= 'A' && password[i] <= 'Z') {
            has_upper = 1;
        }
    }
    
    printf("Password strength check:\n");
    printf("Length: %d %s\n", length, (length >= 8) ? "(Good)" : "(Too short)");
    printf("Has digit: %s\n", has_digit ? "Yes" : "No");
    printf("Has uppercase: %s\n", has_upper ? "Yes" : "No");
    
    return 0;
}
