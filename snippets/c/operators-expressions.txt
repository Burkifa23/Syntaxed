#include <stdio.h>

int main() {
    int a = 15, b = 4, c = 7;
    int result;
    
    printf("Variables: a = %d, b = %d, c = %d\n\n", a, b, c);
    
    // ARITHMETIC OPERATORS
    printf("=== ARITHMETIC OPERATORS ===\n");
    printf("a + b = %d\n", a + b);
    printf("a - b = %d\n", a - b);
    printf("a * b = %d\n", a * b);
    printf("a / b = %d\n", a / b);
    printf("a %% b = %d\n", a % b)
    
    // Float division
    printf("a / (float)b = %.2f\n", a / (float)b);
    
    // INCREMENT/DECREMENT
    printf("\n=== INCREMENT/DECREMENT ===\n");
    int x = 10;
    printf("x = %d\n", x);
    printf("++x = %d\n", ++x);
    printf("x++ = %d\n", x++);
    printf("x = %d\n", x);
    printf("--x = %d\n", --x);
    printf("x-- = %d\n", x--);
    printf("x = %d\n", x);
    
    // COMPARISON OPERATORS
    printf("\n=== COMPARISON OPERATORS ===\n");
    printf("a == b: %d\n", a == b);
    printf("a != b: %d\n", a != b);
    printf("a > b: %d\n", a > b);
    printf("a < b: %d\n", a < b);
    printf("a >= b: %d\n", a >= b);
    printf("a <= b: %d\n", a <= b);
    
    // LOGICAL OPERATORS
    printf("\n=== LOGICAL OPERATORS ===\n");
    int p = 1, q = 0;  // 1 = true, 0 = false
    printf("p = %d, q = %d\n", p, q);
    printf("p && q: %d\n", p && q);
    printf("p || q: %d\n", p || q);
    printf("!p: %d\n", !p);
    printf("!q: %d\n", !q);
    
    // COMPOUND ASSIGNMENT OPERATORS
    printf("\n=== COMPOUND ASSIGNMENT ===\n");
    result = 20;
    printf("Initial result: %d\n", result);
    result += 5;  printf("After += 5: %d\n", result);
    result -= 3;  printf("After -= 3: %d\n", result);
    result *= 2;  printf("After *= 2: %d\n", result);
    result /= 4;  printf("After /= 4: %d\n", result);
    result %= 3;  printf("After %%= 3: %d\n", result);
    
    // OPERATOR PRECEDENCE EXAMPLES
    printf("\n=== OPERATOR PRECEDENCE ===\n");
    printf("2 + 3 * 4 = %d\n", 2 + 3 * 4);
    printf("(2 + 3) * 4 = %d\n", (2 + 3) * 4);
    printf("10 > 5 && 3 < 7 = %d\n", 10 > 5 && 3 < 7);
    
    return 0;
}
