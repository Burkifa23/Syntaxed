#include <stdio.h>

int main() {

    // Declaration and initialization of a 3x3 matrix
    int matrix[3][3] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };

    printf("3x3 Matrix:\n");
    
    // Nested loop to iterate and print the elements
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            printf("%3d", matrix[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}
