#include <stdio.h>

int main(){
    int choice;
    printf("Choose an option:\n");
    printf("1. Add\n2. Subtract\n3. Multiply\n4. Divide\n");
    printf("Enter choice (1-4): ");
    scanf("%d", &choice);
    
    int num1 = 10, num2 = 5;
    float result;
    
    switch (choice) {
        case 1:
            result = num1 + num2;
            printf("%d + %d = %.1f\n", num1, num2, result);
            break;
        case 2:
            result = num1 - num2;
            printf("%d - %d = %.1f\n", num1, num2, result);
            break;
        case 3:
            result = num1 * num2;
            printf("%d * %d = %.1f\n", num1, num2, result);
            break;
        case 4:
            if (num2 != 0) {
                result = (float)num1 / num2;
                printf("%d / %d = %.2f\n", num1, num2, result);
            } else {
                printf("Error: Division by zero!\n");
            }
            break;
        default:
            printf("Invalid choice!\n");
            break;
    }
    
    return 0;

}
